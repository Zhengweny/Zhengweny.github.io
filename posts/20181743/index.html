多次验证网络脚本：将批量IP写入1.txt文件

vi network_check.sh

#!/bin/bash

# 检查参数

if [ $# -lt 2 ]; then

echo "用法: $0 <IP列表文件> <端口1> [端口2] ..."

echo "示例: $0 1.txt 3306 8888 16700"

exit 1

fi

ip_file="$1"

shift

ports=("$@")

# 检查工具是否存在

check_tool() {

if ! command -v "$1" &>/dev/null; then

echo "错误: 未安装 $1，请先安装"

exit 1

fi

}

check_tool nc

check_tool telnet

check_tool timeout

# 测试单个端口的函数

test_port() {

local ip="$1"

local port="$2"

# 方案1: 使用 nc 快速检测

if nc -zv -w 2 "$ip" "$port" &>/dev/null; then

echo "✅ 通 (nc)"

return 0

fi

# 方案2: 使用 telnet 二次验证（避免 nc 误判）

if timeout 3 telnet "$ip" "$port" &>/dev/null <<< EOF; then

echo "✅ 通 (telnet)"

return 0

fi

# 方案3: 使用 /dev/tcp 最终检查（纯 bash）

if timeout 3 bash -c "echo >/dev/tcp/$ip/$port" &>/dev/null; then

echo "✅ 通 (dev/tcp)"

return 0

fi

# 所有方法均失败

echo "❌ 不通"

return 1

}

# 主循环

while read -r ip; do

[[ -z "$ip" || "$ip" =~ ^# ]] && continue

echo "测试 IP: $ip"

for port in "${ports[@]}"; do

echo -n " 端口 $port: "

test_port "$ip" "$port"

done

echo "----------------------"

done < "$ip_file"

使用：./network_check.sh 1.txt 3306 8888 16700 16600 端口4 端口5 ...
